// Magic Download Function (Text ko file banata hai) function downloadCode(elementId, fileName) { // Box ke andar se code nikalna const codeText = document.getElementById(elementId).innerText; // Text ko ek virtual file (Blob) mein convert karna const blob = new Blob([codeText], { type: "text/plain" }); // Download trigger karna const link = document.createElement("a"); link.href = URL.createObjectURL(blob); link.download = fileName; document.body.appendChild(link); link.click(); document.body.removeChild(link); }
